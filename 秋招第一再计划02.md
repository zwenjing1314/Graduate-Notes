# 2）这份配置各部分是干什么的
## `build-system`
这是 Python 打包工具推荐声明的内容，用来告诉工具“怎么构建这个项目”。对新项目来说，写上最稳。([Python Packaging](https://packaging.python.org/en/latest/guides/writing-pyproject-toml/?utm_source=chatgpt.com))

## `[project]`
这里是项目元数据，像名字、版本、Python 版本要求、依赖。PyPA 的 `pyproject.toml` 指南和规范里都把这些作为核心部分。([Python Packaging](https://packaging.python.org/en/latest/guides/writing-pyproject-toml/?utm_source=chatgpt.com))

## `[tool.setuptools]`
因为你现在走的是 `src/` 布局，所以要告诉 setuptools 代码包在 `src` 下面。PyPA 专门讨论过 `src layout` 的做法。([Python Packaging](https://packaging.python.org/en/latest/guides/writing-pyproject-toml/?utm_source=chatgpt.com))

## `[tool.ruff]`
Ruff 官方文档明确支持通过 `pyproject.toml` 配置。这里我给你选的是一套很适合新项目的基础规则：

+ `E` / `F`：基础语法和风格错误
+ `I`：import 排序
+ `B`：一些常见 bug 风险
+ `UP`：更现代的 Python 语法建议

我故意没给你开特别重的规则，不然你第一周会被 lint 折磨。([Astral Docs](https://docs.astral.sh/ruff/configuration/?utm_source=chatgpt.com))

## `[tool.mypy]`
mypy 官方文档说明它会读取 `pyproject.toml`；这里我给你的是“中等严格”配置：

```python
from __future__ import annotations

from dataclasses import dataclass, field
from pathlib import Path
from typing import Any, Literal

DatasetName = Literal["docvqa", "chartqa", "unknown"]
SplitName = Literal["train", "val", "test", "unknown"]


@dataclass(slots=True)
class BBox:
    """Axis-aligned bounding box."""

    x0: float
    y0: float
    x1: float
    y1: float

    def __post_init__(self) -> None:
        if self.x1 < self.x0:
            raise ValueError(f"Invalid bbox: x1({self.x1}) < x0({self.x0})")
        if self.y1 < self.y0:
            raise ValueError(f"Invalid bbox: y1({self.y1}) < y0({self.y0})")

    @property
    def width(self) -> float:
        return self.x1 - self.x0

    @property
    def height(self) -> float:
        return self.y1 - self.y0

    @property
    def area(self) -> float:
        return self.width * self.height

    def as_tuple(self) -> tuple[float, float, float, float]:
        return (self.x0, self.y0, self.x1, self.y1)


@dataclass(slots=True)
class OCRWord:
    """Single OCR word/token."""

    text: str
    bbox: BBox
    confidence: float | None = None
    word_id: str | None = None
    index: int | None = None

    def __post_init__(self) -> None:
        self.text = self.text.strip()

    @property
    def is_empty(self) -> bool:
        return not self.text


@dataclass(slots=True)
class OCRLine:
    """Single OCR text line."""

    text: str
    bbox: BBox
    words: list[OCRWord] = field(default_factory=list)
    line_id: str | None = None
    index: int | None = None

    def __post_init__(self) -> None:
        self.text = self.text.strip()

    @property
    def n_words(self) -> int:
        return len(self.words)

    def reconstructed_text(self) -> str:
        if self.words:
            joined = " ".join(word.text for word in self.words if word.text)
            return joined.strip()
        return self.text


@dataclass(slots=True)
class OCRPage:
    """OCR result for one page/image."""

    page_no: int | None = None
    width: int | None = None
    height: int | None = None
    lines: list[OCRLine] = field(default_factory=list)
    words: list[OCRWord] = field(default_factory=list)
    source: str | None = None  # e.g. "official_ocr", "paddleocr"

    @property
    def n_lines(self) -> int:
        return len(self.lines)

    @property
    def n_words(self) -> int:
        if self.words:
            return len(self.words)
        return sum(line.n_words for line in self.lines)

    def full_text(self) -> str:
        if self.lines:
            return "\n".join(line.reconstructed_text() for line in self.lines if line.reconstructed_text())
        if self.words:
            return " ".join(word.text for word in self.words if word.text)
        return ""


@dataclass(slots=True)
class DocumentSample:
    """
    Canonical sample used inside the project.

    This is intentionally dataset-agnostic:
    - DocVQA can map directly to it
    - ChartQA can also be adapted into it later
    """

    dataset_name: DatasetName
    split: SplitName
    question_id: str
    question: str
    image_path: str
    answers: list[str] = field(default_factory=list)

    # Optional metadata
    doc_id: str | None = None
    document_id: str | None = None
    page_no: int | None = None

    # Optional OCR / structured side info
    ocr_page: OCRPage | None = None
    table_path: str | None = None

    # Keep anything dataset-specific here for forward compatibility
    meta: dict[str, Any] = field(default_factory=dict)

    def __post_init__(self) -> None:
        self.question_id = str(self.question_id).strip()
        self.question = self.question.strip()
        self.image_path = str(self.image_path).strip()

        cleaned_answers: list[str] = []
        for ans in self.answers:
            if isinstance(ans, str):
                ans = ans.strip()
                if ans:
                    cleaned_answers.append(ans)
        self.answers = cleaned_answers

        if not self.question_id:
            raise ValueError("question_id must not be empty")
        if not self.question:
            raise ValueError("question must not be empty")
        if not self.image_path:
            raise ValueError("image_path must not be empty")

    @property
    def primary_answer(self) -> str | None:
        return self.answers[0] if self.answers else None

    @property
    def normalized_answers(self) -> set[str]:
        return {answer.strip().lower() for answer in self.answers if answer.strip()}

    @property
    def has_answers(self) -> bool:
        return len(self.answers) > 0

    @property
    def has_ocr(self) -> bool:
        return self.ocr_page is not None

    @property
    def image_name(self) -> str:
        return Path(self.image_path).name


@dataclass(slots=True)
class DatasetManifest:
    """
    Internal dataset bundle.

    DocVQA official annotations expose dataset-level metadata such as
    dataset_name / dataset_split / dataset_version, so we mirror that idea here.
    """

    dataset_name: DatasetName
    split: SplitName
    version: str = "unknown"
    samples: list[DocumentSample] = field(default_factory=list)
    source_files: dict[str, str] = field(default_factory=dict)

    def __len__(self) -> int:
        return len(self.samples)

    def add(self, sample: DocumentSample) -> None:
        self.samples.append(sample)


@dataclass(slots=True)
class PredictionRecord:
    """One prediction produced by a pipeline."""

    question_id: str
    prediction: str
    dataset_name: DatasetName = "unknown"
    meta: dict[str, Any] = field(default_factory=dict)

    def __post_init__(self) -> None:
        self.question_id = str(self.question_id).strip()
        self.prediction = self.prediction.strip()
        if not self.question_id:
            raise ValueError("question_id must not be empty")
```

这份 schema 的设计思路很简单：

+ `BBox / OCRWord / OCRLine / OCRPage`：给 OCR 和版面理解留好位置
+ `DocumentSample`：统一 DocVQA 和后续 ChartQA 的内部表示
+ `DatasetManifest`：对应 DocVQA 那种“顶层 metadata + data 列表”的组织方式
+ `PredictionRecord`：给后面 `eval.py` 和保存预测结果用

这样做的好处是：**你后面 parser 无论吃 DocVQA 还是 ChartQA，最后都只要产出 **`**DocumentSample**`。DocVQA 官方格式比较稳定，OCR 里也确实包含 line/word 文本和 bounding boxes，所以这套内部 canonical schema 是合理的。 ([docvqa.org](https://www.docvqa.org/datasets))
